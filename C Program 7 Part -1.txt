//Increment and Decrement Operators in C//

INPUT:
#include <stdio.h>

int main()
{
  int a=1;
  printf("a's value is now %d\n", a++);
  printf("a's value is now %d\n", a);
  a= 1;
  printf("a's value is now %d\n", ++a);
  printf("a's value is now %d\n", a);
  a= 1;
  printf("a's value is now %d\n", a--);
  printf("a's value is now %d\n", a);
  a= 1;
  printf("a's value is now %d\n", --a);
  printf("a's value is now %d\n", a);
    return 0;
}

OUTPUT:
a's value is now 1
a's value is now 0
a's value is now 0
a's value is now 0
